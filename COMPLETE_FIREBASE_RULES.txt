rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ✅ Helper: check if user is logged in
    function isAuthed() {
      return request.auth != null;
    }
    
    // ✅ Helper: fetch the current user's role from Firestore (with error handling)
    function userRole() {
      let userDoc = get(/databases/$(database)/documents/users/$(request.auth.uid));
      return userDoc != null && userDoc.data.role != null 
        ? userDoc.data.role 
        : null;
    }
    
    // ✅ Helper: allow STAFF or ADMIN (any capitalization)
    function isStaffOrAdmin() {
      return isAuthed() && 
        (userRole() in ['STAFF', 'ADMIN', 'staff', 'admin', 'Staff', 'Admin']);
    }
    
    // ✅ Helper: allow ADMIN only
    function isAdmin() {
      return isAuthed() && 
        (userRole() in ['ADMIN', 'admin', 'Admin']);
    }
    
    // ---- USERS Collection ----
    match /users/{userId} {
      // ✅ CRITICAL: Allow creating user document (for first-time login)
      allow create: if isAuthed() && request.auth.uid == userId;
      
      // Users can read and update their own document
      allow read, update: if isAuthed() && request.auth.uid == userId;
      
      // Admin (any casing) can read AND update all user profiles
      allow read, update: if isAdmin();
    }
    
    // ---- INVENTORY ----
    match /inventory/{itemId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // ---- CATEGORIES ----
    match /categories/{categoryId} {
      allow read: if isAuthed();
      allow write: if isAdmin(); // Only admins can manage categories
    }
    
    // ---- DOCUMENTS ----
    match /documents/{documentId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // ---- DELIVERIES ----
    match /deliveries/{deliveryId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // ---- DELIVERY LOGS (Mobile App Status Updates) ----
    match /delivery_logs/{logId} {
      allow read: if isAuthed();
      allow write: if isAuthed(); // Allow delivery staff to update status
    }
    
    // ---- STOCK MOVEMENTS ----
    match /stock_movements/{movementId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // ---- STOCK LOGS ----
    match /stock_logs/{logId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // ---- CUSTOMER TRANSACTIONS ----
    match /customer_transactions/{transactionId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // ---- LOCATIONS ----
    match /locations/{locationId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // ---- QC REPORTS ----
    match /qc_reports/{reportId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // ---- EXPIRATION ALERTS ----
    match /expiration_alerts/{alertId} {
      allow read: if isAuthed();
      allow write: if isStaffOrAdmin();
    }
    
    // Default: deny everything else
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
